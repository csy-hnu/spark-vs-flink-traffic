一、创建数据库与表
-- 创建数据库
CREATE DATABASE traffic_db;
USE traffic_db;

-- 对比一：小时级峰值检测结果表
CREATE TABLE IF NOT EXISTS batch_hourly_baseline (
    is_holiday INT,
    hour INT,
    avg_traffic DOUBLE
);

-- 对比二：日均流量结果表
CREATE TABLE IF NOT EXISTS batch_daily_average (
    date VARCHAR(20),
    is_holiday INT,
    avg_daily_traffic DOUBLE
);

-- 对比三：累计流量结果表
CREATE TABLE IF NOT EXISTS batch_daily_total (
    date VARCHAR(20),
    is_holiday INT,
    total_daily_traffic BIGINT
);

-- Flink 对比一：告警结果表
CREATE TABLE IF NOT EXISTS flink_alert_result (
    date_time VARCHAR(30),
    traffic INT,
    baseline DOUBLE,
    is_holiday INT,
    hour INT,
    alert_status VARCHAR(10)
);

-- Flink 对比二：滚动均值结果表
CREATE TABLE IF NOT EXISTS flink_rolling_avg_result (
    current_time VARCHAR(50),
    window_start VARCHAR(50),
    avg_traffic DOUBLE,
    data_count INT
);

-- Flink 对比三：累计流量结果表
CREATE TABLE IF NOT EXISTS flink_cumulative_result (
    date VARCHAR(20),
    hour INT,
    cumulative_traffic BIGINT,
    is_holiday INT
);

-- 查看所有表
SHOW TABLES;

三、结果正确性验证
3.1 验证节假日与非节假日流量差异
SELECT 
    b1.hour,
    b1.avg_traffic AS non_holiday_baseline,
    b2.avg_traffic AS holiday_baseline,
    ROUND((b2.avg_traffic - b1.avg_traffic) / b1.avg_traffic * 100, 2) AS diff_percent
FROM batch_hourly_baseline b1
JOIN batch_hourly_baseline b2 ON b1.hour = b2.hour
WHERE b1.is_holiday = 0 AND b2.is_holiday = 1
ORDER BY b1.hour;

3.2 验证告警阈值正确性
-- 查看告警状态分布
SELECT alert_status, COUNT(*) FROM flink_alert_result GROUP BY alert_status;

-- 验证 HIGH_ALERT 是否都 >= 1.5 倍
SELECT COUNT(*) AS invalid_count
FROM flink_alert_result a
JOIN batch_hourly_baseline b
ON a.is_holiday = b.is_holiday AND a.hour = b.hour
WHERE a.alert_status = 'HIGH_ALERT'
AND a.traffic / b.avg_traffic < 1.5;

-- 验证 LOW_ALERT 是否都 <= 0.5 倍
SELECT COUNT(*) AS invalid_count
FROM flink_alert_result a
JOIN batch_hourly_baseline b
ON a.is_holiday = b.is_holiday AND a.hour = b.hour
WHERE a.alert_status = 'LOW_ALERT'
AND a.traffic / b.avg_traffic > 0.5;

-- 验证 NORMAL 是否都在正常范围内（0.5 ~ 1.5 倍）
SELECT 
    COUNT(*) AS total_normal,
    SUM(CASE WHEN traffic / avg_traffic < 0.5 THEN 1 ELSE 0 END) AS below_low_threshold,
    SUM(CASE WHEN traffic / avg_traffic > 1.5 THEN 1 ELSE 0 END) AS above_high_threshold,
    SUM(CASE WHEN traffic / avg_traffic BETWEEN 0.5 AND 1.5 THEN 1 ELSE 0 END) AS within_range
FROM flink_alert_result a
JOIN batch_hourly_baseline b
ON a.is_holiday = b.is_holiday AND a.hour = b.hour
WHERE a.alert_status = 'NORMAL';

3.3 验证累计终点一致性
SELECT 
    '非节假日' AS day_type,
    s.total_daily_traffic AS spark_total,
    f.cumulative_traffic AS flink_final,
    s.total_daily_traffic - f.cumulative_traffic AS diff
FROM batch_daily_total s
JOIN flink_cumulative_result f
ON s.date = f.date
WHERE s.date = '2012-10-10' AND f.hour = 23
UNION ALL
SELECT 
    '节假日' AS day_type,
    s.total_daily_traffic AS spark_total,
    f.cumulative_traffic AS flink_final,
    s.total_daily_traffic - f.cumulative_traffic AS diff
FROM batch_daily_total s
JOIN flink_cumulative_result f
ON s.date = f.date
WHERE s.date = '2012-12-25' AND f.hour = 23;

3.4 验证滚动均值收敛性
SELECT 
    r.current_time,
    r.avg_traffic AS flink_rolling_avg,
    d.avg_daily_traffic AS spark_daily_avg,
    ROUND(ABS(r.avg_traffic - d.avg_daily_traffic), 2) AS error,
    ROUND(ABS(r.avg_traffic - d.avg_daily_traffic) / d.avg_daily_traffic * 100, 2) AS error_pct
FROM flink_rolling_avg_result r
JOIN batch_daily_average d ON DATE(r.current_time) = d.date
WHERE d.date = '2012-10-10'
ORDER BY r.current_time;
